Step 1: Run local frontmatter pre-validation for DSC-03 and DSC-04.

Run the following Python validation script to verify all 4 skills pass the name regex and directory-name matching:

```bash
python3 -c "
import yaml, os, re, glob
regex = re.compile(r'^[a-z0-9]+(-[a-z0-9]+)*\$')
all_pass = True
for path in sorted(glob.glob('skills/*/SKILL.md')):
    dirname = os.path.basename(os.path.dirname(path))
    with open(path) as f:
        parts = f.read().split('---', 2)
    fm = yaml.safe_load(parts[1])
    name = fm.get('name', '')
    desc = fm.get('description', '')
    name_valid = bool(regex.match(name))
    dir_match = name == dirname
    desc_valid = 1 <= len(desc) <= 1024
    status = 'PASS' if all([name_valid, dir_match, desc_valid]) else 'FAIL'
    if status == 'FAIL':
        all_pass = False
    print(f'{status}: {dirname} (name={name}, name_valid={name_valid}, dir_match={dir_match}, desc_len={len(desc)})')
if all_pass:
    print('\nAll 4 skills pass local pre-validation (DSC-03, DSC-04)')
else:
    print('\nFAILURE: One or more skills failed pre-validation')
    exit(1)
"
```
