```python
"""Tier 4 live e2e test — 3-paper + 5-image subset.

Filename matches Phase 129 Plan 07 convention (test_tierN_e2e_live.py) for pytest
rootdir basename uniqueness (no `__init__.py` per Phase 128 Plan 02 follow-on rule).

Cost: ~$0.10-0.30 per run. Pre-condition: OPENROUTER_API_KEY in env.
First run also triggers MineRU model download (~3-5 GB; ~5-15 min cold start).
"""
import asyncio
import json
import os
import shutil
import time
from pathlib import Path

import pytest

from shared.config import get_settings
from shared.cost_tracker import CostTracker
from shared.loader import DatasetLoader

from tier_4_multimodal.rag import build_rag, DEFAULT_LLM_MODEL, DEFAULT_EMBED_MODEL
from tier_4_multimodal.cost_adapter import CostAdapter
from tier_4_multimodal.ingest_pdfs import ingest_pdfs
from tier_4_multimodal.ingest_images import ingest_standalone_images
from tier_4_multimodal.query import run_query


SUBSET_PAPERS = 3
SUBSET_IMAGES = 5
QUESTION = (
    "Summarize the main contribution of the provided documents in 2-3 sentences. "
    "If figures are referenced, mention what the figure depicts."
)


@pytest.mark.live
def test_tier4_end_to_end_subset(tier4_live_keys_ok, tmp_path):
    """End-to-end Tier 4 — 3 PDFs + 5 images via hybrid mode, cost > 0, KG files persisted."""
    settings = get_settings()
    dataset_root = Path(settings.dataset_root)

    # Pick 3 SMALLEST papers (cost-bound; mirrors Phase 129 Plan 07's subset choice).
    loader = DatasetLoader(dataset_root)
    all_papers = loader.papers()
    sized = [
        (p, (dataset_root / "papers" / p["filename"]).stat().st_size)
        for p in all_papers
        if (dataset_root / "papers" / p["filename"]).exists()
    ]
    sized.sort(key=lambda x: x[1])
    papers_subset = [p for p, _ in sized[:SUBSET_PAPERS]]
    assert len(papers_subset) == SUBSET_PAPERS, "Need ≥3 papers in dataset/papers/"

    # Build a synthetic figures manifest from the first 5 entries of the real one
    # (ABSOLUTE img_path — Pitfall 4 mitigation already applied by ingest_images).
    figures_path = dataset_root / "manifests" / "figures.json"
    if figures_path.exists():
        all_figs = json.loads(figures_path.read_text())[:SUBSET_IMAGES]
    else:
        all_figs = []
    fake_dataset = tmp_path / "dataset_subset"
    fake_dataset.mkdir()
    (fake_dataset / "manifests").mkdir()
    (fake_dataset / "images").mkdir()
    # Symlink (or copy) the 5 image files into our subset
    real_images = dataset_root / "images"
    subset_figs = []
    for f in all_figs:
        src = real_images / f["filename"]
        if not src.exists():
            continue
        dst = fake_dataset / "images" / f["filename"]
        try:
            os.symlink(src, dst)
        except OSError:
            shutil.copyfile(src, dst)
        subset_figs.append(f)
    (fake_dataset / "manifests" / "figures.json").write_text(json.dumps(subset_figs))

    # Working dir under tmp_path so we don't pollute user state
    working_dir = str(tmp_path / "rag_anything_storage" / "tier-4-multimodal-test")

    rag = build_rag(working_dir=working_dir)
    tracker = CostTracker("tier-4-test")
    rag.llm_token_tracker = CostAdapter(
        tracker,
        llm_model=DEFAULT_LLM_MODEL,
        embed_model=DEFAULT_EMBED_MODEL,
    )

    # --- Ingest ---
    t0 = time.monotonic()
    asyncio.run(_ingest(rag, papers_subset, dataset_root, fake_dataset))
    ingest_latency = time.monotonic() - t0
    print(f"\nIngest latency: {ingest_latency:.2f}s")

    # --- Query ---
    t0 = time.monotonic()
    answer = asyncio.run(run_query(rag, QUESTION, mode="hybrid"))
    query_latency = time.monotonic() - t0
    print(f"Query latency: {query_latency:.2f}s")
    print(f"Answer: {answer[:300]}{'...' if len(str(answer)) > 300 else ''}")

    # --- Assertions ---
    # 1. Working dir created with KG file
    graphml = Path(working_dir) / "graph_chunk_entity_relation.graphml"
    assert graphml.exists(), f"KG file missing at {graphml}"
    # NetworkX placeholder is ~tens of bytes; a real ingest produces multi-KB.
    assert graphml.stat().st_size > 1024, f"KG file suspiciously small: {graphml.stat().st_size}B"

    # 2. Answer is non-empty
    assert isinstance(answer, str) and len(answer.strip()) > 0, "Empty answer"

    # 3. Cost was tracked
    assert tracker.total_usd() > 0, "Cost tracker recorded zero — token_tracker not wired?"
    print(f"Total cost: ${tracker.total_usd():.6f}")

    tracker.persist()


async def _ingest(rag, papers, dataset_root, fake_dataset):
    from rich.console import Console
    console = Console()
    await ingest_pdfs(rag, papers, dataset_root, console, device="cpu", parser="mineru")
    await ingest_standalone_images(rag, fake_dataset)
```
