`tier-4-multimodal/query.py` (Pattern 4):

```python
"""Tier 4 — async query wrapper with multimodal-content branching."""
from __future__ import annotations
from typing import Optional


async def run_query(rag, question: str, mode: str = "hybrid",
                    multimodal_content: Optional[list] = None) -> str:
    """Run a Tier 4 query.

    If `multimodal_content` is supplied (a list of {type,...} dicts per the
    RAG-Anything README), uses `aquery_with_multimodal`. Else uses plain `aquery` —
    RAG-Anything still surfaces image/table chunks because they are indexed in
    the same multimodal KG.
    """
    if multimodal_content:
        return await rag.aquery_with_multimodal(
            question, multimodal_content=multimodal_content, mode=mode
        )
    return await rag.aquery(question, mode=mode)


def to_display_chunks(response) -> list[dict]:
    """Defensive adapter from RAG-Anything response → shared.display chunks shape.

    RAG-Anything 1.2.10 returns a string answer (Pitfall 7) — there is no
    structured `chunks` field exposed publicly. We pass [] and let the agent
    self-cite in the answer text. If a future raganything version DOES expose
    `response.context` or similar, populate from there; until then, return [].
    """
    # Defensive: if some future version returns a dict, look for a chunks/context key.
    if isinstance(response, dict):
        for key in ("chunks", "context", "retrieved_chunks"):
            if key in response and isinstance(response[key], list):
                return [
                    {
                        "doc_id": (c or {}).get("doc_id") or (c or {}).get("source") or "",
                        "score": float((c or {}).get("score") or 0.0),
                        "snippet": str((c or {}).get("text") or (c or {}).get("snippet") or "")[:400],
                    }
                    for c in response[key] if c
                ]
    return []
```

`tier-4-multimodal/main.py` (mirror Phase 129 Plan 05's structure: argparse + asyncio.run + cost-surprise gate + device autodetect):

```python
"""Tier 4 — Multimodal RAG CLI (RAG-Anything).

Run:
    python tier-4-multimodal/main.py                 # default: ingest + canned query
    python tier-4-multimodal/main.py --query "..."   # query-only against existing working_dir
    python tier-4-multimodal/main.py --ingest --no-images   # PDFs only, skip standalone images
    python tier-4-multimodal/main.py --reset --ingest --yes # nuke + re-ingest (cost-surprise gate)
"""
from __future__ import annotations
import argparse
import asyncio
import shutil
import sys
import time
from pathlib import Path

# Repo-root sys.path bootstrap so `from shared.* import *` works regardless of CWD.
_REPO_ROOT = Path(__file__).resolve().parent.parent
if str(_REPO_ROOT) not in sys.path:
    sys.path.insert(0, str(_REPO_ROOT))

from rich.console import Console

from shared.config import get_settings
from shared.cost_tracker import CostTracker
from shared.display import render_query_result
from shared.loader import DatasetLoader
from tier_4_multimodal.rag import (
    build_rag,
    WORKING_DIR,
    DEFAULT_LLM_MODEL,
    DEFAULT_VISION_MODEL,
    DEFAULT_EMBED_MODEL,
)
from tier_4_multimodal.cost_adapter import CostAdapter
from tier_4_multimodal.ingest_pdfs import ingest_pdfs
from tier_4_multimodal.ingest_images import ingest_standalone_images
from tier_4_multimodal.query import run_query, to_display_chunks


DEFAULT_QUERY = (
    "What does Figure 1 of the Attention Is All You Need paper depict, "
    "and how does the encoder-decoder architecture relate to retrieval-augmented generation?"
)


def _detect_device() -> str:
    """Open Q3 — print detected device at ingest start. Falls back to cpu."""
    try:
        import torch
        if torch.cuda.is_available():
            return "cuda:0"
        if hasattr(torch.backends, "mps") and torch.backends.mps.is_available():
            return "mps"
    except Exception:
        pass
    return "cpu"


def _confirm_or_abort(prompt: str, *, auto_yes: bool) -> bool:
    """Cost-surprise gate. CI-safe: returns False on EOF/empty (Phase 129 Plan 05 pattern)."""
    if auto_yes:
        return True
    try:
        ans = input(prompt).strip().lower()
    except EOFError:
        return False
    return ans in {"y", "yes"}


async def cmd_ingest(rag, settings, console: Console, args, tracker: CostTracker) -> int:
    dataset_root = Path(settings.dataset_root)
    loader = DatasetLoader(dataset_root)
    papers = loader.papers()

    n_pdfs = 0
    if args.include_pdfs:
        n_pdfs = await ingest_pdfs(
            rag, papers, dataset_root, console,
            device=args.device, parser="mineru",
        )

    n_images = 0
    if args.include_images:
        n_images = await ingest_standalone_images(rag, dataset_root)

    console.print(
        f"[green]Tier 4 ingest complete: {n_pdfs} PDFs + {n_images} images.[/green]"
    )
    return 0


async def cmd_query(rag, console: Console, args, tracker: CostTracker) -> int:
    question = args.query or DEFAULT_QUERY
    t0 = time.monotonic()
    answer = await run_query(rag, question, mode=args.mode)
    latency = time.monotonic() - t0

    chunks = to_display_chunks(answer if not isinstance(answer, str) else None)

    # CostAdapter has accumulated LLM + embed token usage during ingest+query.
    # tracker.total_usd() reflects the running total.
    in_tok = sum((line.get("input_tokens") or 0) for line in tracker.lines if (line or {}).get("kind") == "llm")
    out_tok = sum((line.get("output_tokens") or 0) for line in tracker.lines if (line or {}).get("kind") == "llm")

    render_query_result(
        query=question,
        chunks=chunks,
        answer=str(answer),
        cost_usd=tracker.total_usd(),
        input_tokens=in_tok,
        output_tokens=out_tok,
        console_override=console,
    )
    console.print(f"[dim]Latency: {latency:.2f}s, mode={args.mode}[/dim]")
    return 0


async def amain(args) -> int:
    settings = get_settings()
    console = Console(record=True)

    if not settings.openrouter_api_key:
        console.print("[red]OPENROUTER_API_KEY is not set — Tier 4 cannot run.[/red]")
        console.print("[red]See tier-4-multimodal/README.md → Quickstart.[/red]")
        return 2

    detected_device = _detect_device()
    if args.device == "auto":
        args.device = detected_device
    console.print(f"[dim]Device: {args.device} (autodetected: {detected_device})[/dim]")

    if args.reset:
        if Path(WORKING_DIR).exists():
            if not _confirm_or_abort(
                f"--reset will delete {WORKING_DIR}. Continue? [y/N] ",
                auto_yes=args.yes,
            ):
                console.print("[yellow]Aborted.[/yellow]")
                return 1
            shutil.rmtree(WORKING_DIR)

    if args.ingest:
        if not Path(WORKING_DIR, "graph_chunk_entity_relation.graphml").exists() or args.reset:
            if not _confirm_or_abort(
                "Tier 4 ingest will run MineRU + entity extraction "
                "(~$0.50-1.50 for full corpus, ~$0.05 for 3-paper subset). Continue? [y/N] ",
                auto_yes=args.yes,
            ):
                console.print("[yellow]Aborted.[/yellow]")
                return 1

    rag = build_rag(working_dir=WORKING_DIR)
    tracker = CostTracker("tier-4")
    rag.llm_token_tracker = CostAdapter(
        tracker,
        llm_model=args.model or DEFAULT_LLM_MODEL,
        embed_model=DEFAULT_EMBED_MODEL,
    )

    if args.ingest:
        rc = await cmd_ingest(rag, settings, console, args, tracker)
        if rc != 0:
            return rc

    rc = await cmd_query(rag, console, args, tracker)
    tracker.persist()
    console.print(f"[dim]Cost JSON: evaluation/results/costs/tier-4-{tracker.timestamp}.json[/dim]")
    return rc


def build_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(description="Tier 4 — Multimodal RAG via RAG-Anything.")
    p.add_argument("--ingest", action="store_true", help="Run ingest pass (PDFs + images).")
    p.add_argument("--query", default=None, help="Question to answer. Defaults to a canned multimodal probe.")
    p.add_argument("--mode", default="hybrid", choices=["naive", "local", "global", "hybrid", "mix"])
    p.add_argument("--model", default=None, help=f"OpenRouter LLM slug (default: {DEFAULT_LLM_MODEL}).")
    p.add_argument("--reset", action="store_true", help="Delete the working_dir before ingest.")
    p.add_argument("--include-images", dest="include_images", action="store_true", default=True)
    p.add_argument("--no-images", dest="include_images", action="store_false")
    p.add_argument("--include-pdfs", dest="include_pdfs", action="store_true", default=True)
    p.add_argument("--no-pdfs", dest="include_pdfs", action="store_false")
    p.add_argument("--device", default="auto", help="cpu, cuda:0, mps, or auto (default).")
    p.add_argument("--yes", action="store_true", help="Auto-confirm cost-surprise prompts (CI use).")
    return p


def main(argv: list[str] | None = None) -> int:
    args = build_parser().parse_args(argv)
    if not args.ingest and not args.query:
        # Phase 128 Plan 04 default-flag-less convention: ingest + canned query
        args.ingest = True
    return asyncio.run(amain(args))


if __name__ == "__main__":
    raise SystemExit(main())
```
