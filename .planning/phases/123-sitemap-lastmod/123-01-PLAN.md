```typescript
import { execSync } from 'node:child_process';

const cache = new Map<string, string | undefined>();

/**
 * Return the committer-ISO-8601 date of the most recent commit that touched
 * `relativePath` (relative to repo root), normalized to UTC with trailing 'Z'.
 *
 * Determinism: committer dates are immutable per commit. Two builds on the
 * same commit return the same value. Cached per process.
 *
 * Caveats:
 * - On shallow clones (CI default fetch-depth: 1) the query may return empty
 *   for files whose last-touching commit is outside the shallow window. In
 *   that case this returns undefined — the caller must decide to fall back
 *   or fail. Prefer frontmatter/JSON/registry sources and treat git log as
 *   a last-resort fallback.
 * - Do NOT pass absolute paths — git expects a repo-relative path.
 */
export function gitLogDate(relativePath: string): string | undefined {
  if (cache.has(relativePath)) return cache.get(relativePath);
  let result: string | undefined;
  try {
    const out = execSync(
      `git log -1 --format=%cI -- "${relativePath}"`,
      { encoding: 'utf-8', stdio: ['ignore', 'pipe', 'ignore'] }
    ).trim();
    if (out) {
      result = new Date(out).toISOString();
    }
  } catch {
    result = undefined;
  }
  cache.set(relativePath, result);
  return result;
}
```

**Step 1: Create `src/lib/sitemap/content-dates.ts`**

```typescript
import { readdirSync, readFileSync } from 'node:fs';
import { join, basename, extname } from 'node:path';
import { STATIC_PAGE_DATES, TOOL_RULES_DATES, COLLECTION_SHIP_DATES } from './static-dates';

const SITE = 'https://patrykgolabek.dev';
const DATE_RE = /(?:updatedDate|publishedDate|lastVerified):\s*["']?(\d{4}-\d{2}-\d{2})["']?/g;

function isoFromYmd(ymd: string): string {
  return new Date(ymd + 'T00:00:00Z').toISOString();
}

/**
 * Extract the best available date from MDX/MD frontmatter.
 * Precedence: updatedDate > lastVerified > publishedDate.
 */
function extractFrontmatterDate(raw: string): string | undefined {
  let published = '';
  let updated = '';
  let verified = '';
  for (const m of raw.matchAll(DATE_RE)) {
    if (m[0].startsWith('updated')) updated = m[1];
    else if (m[0].startsWith('lastVerified')) verified = m[1];
    else published = m[1];
  }
  return updated || verified || published || undefined;
}

function isExternalBlog(raw: string): boolean {
  return /^externalUrl:/m.test(raw);
}

export function buildContentDateMap(): Map<string, string> {
  const map = new Map<string, string>();

  // 1. Static page registry
  for (const [url, date] of Object.entries(STATIC_PAGE_DATES)) map.set(url, date);

  // 2. Blog posts — per-post frontmatter (updatedDate || publishedDate).
  //    External-URL posts do NOT generate /blog/{slug}/ routes, so skip them.
  //    (Plan 02 handles blog index/pagination/tag aggregation.)
  const blogDir = './src/data/blog';
  for (const file of readdirSync(blogDir)) {
    if (!/\.(md|mdx)$/.test(file)) continue;
    const raw = readFileSync(join(blogDir, file), 'utf-8');
    if (isExternalBlog(raw)) continue;
    const slug = basename(file, extname(file));
    const date = extractFrontmatterDate(raw);
    if (date) map.set(`${SITE}/blog/${slug}/`, isoFromYmd(date));
  }

  // 3. Guides — per-guide dir, per-chapter MDX frontmatter with guide-JSON fallback.
  //    BUG FIX: previous logic applied guide.json.publishedDate to EVERY chapter,
  //    ignoring per-page frontmatter. Claude Code chapters all set updatedDate
  //    2026-04-12 but sitemap showed 2026-03-15. Now each chapter URL uses its
  //    own frontmatter date first, falling back to guide.json only if missing.
  const guidesDir = './src/data/guides';
  for (const entry of readdirSync(guidesDir, { withFileTypes: true })) {
    if (!entry.isDirectory()) continue;
    const guideDir = `${guidesDir}/${entry.name}`;
    let guideMeta: any;
    try {
      guideMeta = JSON.parse(readFileSync(`${guideDir}/guide.json`, 'utf-8'))[0];
    } catch { continue; }
    if (!guideMeta?.slug) continue;
    const guideSlug = guideMeta.slug;
    const guideFallback = guideMeta.publishedDate ? isoFromYmd(guideMeta.publishedDate) : undefined;

    const chapterDates: string[] = [];
    for (const ch of guideMeta.chapters ?? []) {
      const mdxPath = `${guideDir}/pages/${ch.slug}.mdx`;
      let chDate: string | undefined = guideFallback;
      try {
        const raw = readFileSync(mdxPath, 'utf-8');
        const fm = extractFrontmatterDate(raw);
        if (fm) chDate = isoFromYmd(fm);
      } catch {
        // Chapter MDX not present (e.g., synthetic cheatsheet routes). Keep fallback.
      }
      if (chDate) {
        map.set(`${SITE}/guides/${guideSlug}/${ch.slug}/`, chDate);
        chapterDates.push(chDate);
      }
    }

    // Guide root = max(chapter dates) or guide publishedDate.
    if (chapterDates.length) {
      const sorted = [...chapterDates].sort();
      map.set(`${SITE}/guides/${guideSlug}/`, sorted[sorted.length - 1]);
    } else if (guideFallback) {
      map.set(`${SITE}/guides/${guideSlug}/`, guideFallback);
    }
  }

  // /guides/ landing = max of guide roots.
  const guideRootDates = [...map.entries()]
    .filter(([url]) => /\/guides\/[^/]+\/$/.test(url.replace(SITE, '')))
    .map(([, d]) => d)
    .sort();
  if (guideRootDates.length) map.set(`${SITE}/guides/`, guideRootDates[guideRootDates.length - 1]);

  // 4. Beauty Index language pages — COLLECTION_SHIP_DATES.beautyIndexLanguage
  //    populated per-URL so Phase 125's OPSEO-03 language-description edits can
  //    later bump individual entries without affecting VS or concept dates.
  try {
    const langs = JSON.parse(readFileSync('./src/data/beauty-index/languages.json', 'utf-8'));
    for (const l of langs) {
      if (l?.id) map.set(`${SITE}/beauty-index/${l.id}/`, COLLECTION_SHIP_DATES.beautyIndexLanguage);
    }
  } catch { /* non-fatal */ }

  // 5. Beauty Index VS pages — 650 pairs all stamped with Phase 122 enrichment ship date.
  //    Pairs are derived from C(28, 2) where both languages exist and are
  //    canonicalised alpha-order {a}-vs-{b}. We iterate the same sorted list
  //    the page route does. (Deterministic: sorted JSON IDs.)
  try {
    const langs = JSON.parse(readFileSync('./src/data/beauty-index/languages.json', 'utf-8'));
    const ids = langs.map((l: any) => l.id).filter(Boolean).sort();
    for (let i = 0; i < ids.length; i++) {
      for (let j = i + 1; j < ids.length; j++) {
        const slug = `${ids[i]}-vs-${ids[j]}`;
        map.set(`${SITE}/beauty-index/vs/${slug}/`, COLLECTION_SHIP_DATES.beautyIndexVs);
      }
    }
  } catch { /* non-fatal */ }

  // 6. AI Landscape concepts (51) + VS pairs (12) — collection ship date.
  try {
    const nodes = JSON.parse(readFileSync('./src/data/ai-landscape/nodes.json', 'utf-8'));
    for (const n of nodes) {
      if (n?.slug) map.set(`${SITE}/ai-landscape/${n.slug}/`, COLLECTION_SHIP_DATES.aiLandscape);
    }
  } catch { /* non-fatal */ }
  // VS pairs are generated from a static comparisons list inside the route file;
  // for Phase 123 we apply the same collection-wide ship date to whatever URLs
  // the route emits. The serialize hook will pass through each URL; the lookup
  // for /ai-landscape/vs/{slug}/ must resolve even without enumerating pairs
  // here. Handle by registering a URL-prefix default in the serialize hook
  // (see astro.config.mjs edit below) OR enumerate the 12 pairs from a helper.
  // Chosen approach: add a prefix-default in the serialize hook so future pair
  // additions don't silently lose their lastmod.

  // 7. DB Compass models — collection ship date.
  try {
    const models = JSON.parse(readFileSync('./src/data/db-compass/models.json', 'utf-8'));
    for (const m of models) {
      if (m?.id) map.set(`${SITE}/db-compass/${m.id}/`, COLLECTION_SHIP_DATES.dbCompassModel);
    }
  } catch { /* non-fatal */ }

  // 8. Tool rules — per-tool batch date. 213 URLs across 4 tools.
  //    Rules live in src/lib/tools/{tool}/rules/*.ts — we don't enumerate here;
  //    the serialize hook uses a URL-prefix lookup (/tools/{tool}/rules/) to
  //    apply TOOL_RULES_DATES[tool]. See astro.config.mjs edit.

  return map;
}

// Exported for use in astro.config.mjs to handle prefix-based fallbacks
// (ai-landscape/vs, tool rules) where per-URL enumeration would couple the
// sitemap logic to route internals.
export function resolvePrefixLastmod(url: string): string | undefined {
  // /ai-landscape/vs/{pair}/ — collection ship date
  if (url.startsWith(`${SITE}/ai-landscape/vs/`)) return COLLECTION_SHIP_DATES.aiLandscape;
  // /tools/{tool}/rules/{code}/ — per-tool batch date
  const ruleMatch = url.match(new RegExp(`^${SITE.replace(/\./g, '\\.')}/tools/([^/]+)/rules/`));
  if (ruleMatch) return TOOL_RULES_DATES[ruleMatch[1]];
  return undefined;
}
```
