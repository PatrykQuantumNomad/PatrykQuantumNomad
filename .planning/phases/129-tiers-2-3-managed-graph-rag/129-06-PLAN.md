   - Indexing cost is a synthetic estimate (not server-reported). Phase 131 evaluation will report the actual delta.
   - Citations are surface-level — `grounding_metadata.grounding_chunks[i].retrieved_context.title` + `.text` only. Score field may not be exposed by `gemini-2.5-flash` (Open Q3 — defensive `getattr(ctx, "score", 0.0)`).

```python
"""Tier 2 live e2e test — 3-paper subset to dodge Pitfall 2 503 storms."""
import os
import time
from datetime import datetime, timezone
from pathlib import Path

import pytest
from google import genai

from shared.config import get_settings
from shared.cost_tracker import CostTracker
from tier_2_managed.store import (
    upload_with_retry,
    list_existing_documents,
    delete_store,
    STORE_DISPLAY_NAME,
)
from tier_2_managed.query import query as tier2_query, to_display_chunks


SUBSET_PAPERS = 3  # Pitfall 2 — keep test runtime short and dodge TPM saturation


@pytest.mark.live
def test_tier2_end_to_end(tier2_live_keys_ok, tmp_path):
    """End-to-end: create unique store, upload 3 papers, query, assert citations + cost > 0, cleanup."""
    settings = get_settings()
    client = genai.Client(api_key=settings.gemini_api_key.get_secret_value())
    assert hasattr(client, "file_search_stores"), "google-genai >= 1.49 required"

    # Create a UNIQUE store for this test run (don't pollute the user's main store).
    from google.genai import types
    unique_name = f"rag-arch-tier-2-test-{datetime.now(timezone.utc).strftime('%Y%m%d-%H%M%S')}"
    store = client.file_search_stores.create(
        config=types.CreateFileSearchStoreConfig(display_name=unique_name)
    )
    store_name = store.name

    try:
        # Upload 3 papers (subset). Use the smallest by file size for speed.
        from shared.loader import DatasetLoader
        loader = DatasetLoader()
        papers = loader.papers()
        dataset_root = Path(get_settings().dataset_root)
        candidates = [
            (p, (dataset_root / "papers" / p["filename"]).stat().st_size)
            for p in papers
            if (dataset_root / "papers" / p["filename"]).exists()
        ]
        candidates.sort(key=lambda x: x[1])
        subset = [p for p, _ in candidates[:SUBSET_PAPERS]]
        assert len(subset) == SUBSET_PAPERS, "Need at least 3 papers in dataset"

        for p in subset:
            pdf_path = dataset_root / "papers" / p["filename"]
            upload_with_retry(client, store_name, str(pdf_path), p["filename"])

        # Confirm uploads landed.
        existing = list_existing_documents(client, store_name)
        assert len(existing) == SUBSET_PAPERS, f"expected {SUBSET_PAPERS} docs, got {len(existing)}"

        # Run a query. Pick a question relevant to the smallest-papers subset; if the subset
        # is RAG-paper-adjacent, the canned single-hop-001 works; otherwise use a generic prompt.
        question = "Summarize the main contribution of the provided documents in two sentences."
        tracker = CostTracker("tier-2-test")
        t0 = time.monotonic()
        resp = tier2_query(client, store_name, question)
        latency = time.monotonic() - t0

        # Assertions — the three success criteria, distilled.
        assert resp is not None
        assert getattr(resp, "text", None), "Empty answer"
        chunks = to_display_chunks(resp)
        # grounding may be empty if the model thinks the corpus is irrelevant; soft assert
        # (Pitfall 6 — research notes None grounding_metadata is possible).
        # We assert ≥ 0 (always true) but log the count for visibility.
        print(f"\nGrounding chunks: {len(chunks)}, latency {latency:.2f}s")

        # Cost: usage_metadata is the source of truth.
        usage = getattr(resp, "usage_metadata", None)
        in_tok = int(getattr(usage, "prompt_token_count", 0) or 0)
        out_tok = int(getattr(usage, "candidates_token_count", 0) or 0)
        if in_tok or out_tok:
            tracker.record_llm("gemini-2.5-flash", in_tok, out_tok)
        assert tracker.total_usd() > 0, "Cost tracker should record non-zero LLM cost"
    finally:
        delete_store(client, store_name)
```
