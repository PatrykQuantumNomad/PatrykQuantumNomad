cost_adapter.py contract (Pattern 6 with documented fallback for Open Q1):

```python
"""LightRAG token_tracker → shared.cost_tracker.CostTracker bridge.

The exact protocol shape varies across LightRAG versions. This module:
1. Wraps `shared.cost_tracker.CostTracker` in a `CostAdapter` that exposes
   the methods LightRAG's source calls (verified via the probe in this plan's
   Task 1).
2. Documents the monkey-patch fallback in case the probe reveals an API mismatch
   that the adapter cannot cleanly satisfy.
"""

class CostAdapter:
    def __init__(self, tracker, model: str):
        self.tracker = tracker
        self.model = model

    def add_usage(self, usage):
        # Verified protocol on lightrag-hku==1.4.15 (Plan 03 Task 1 probe):
        #   add_usage(usage) where usage has prompt_tokens / completion_tokens
        prompt = self._extract(usage, "prompt_tokens")
        completion = self._extract(usage, "completion_tokens")
        if prompt or completion:
            self.tracker.record_llm(self.model, prompt, completion)

    @staticmethod
    def _extract(usage, key: str) -> int:
        if usage is None:
            return 0
        if hasattr(usage, key):
            return int(getattr(usage, key) or 0)
        if isinstance(usage, dict):
            return int(usage.get(key) or 0)
        return 0
```
