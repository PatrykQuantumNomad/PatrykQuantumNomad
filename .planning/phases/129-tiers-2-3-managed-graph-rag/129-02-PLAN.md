```python
"""Importable shim for the on-disk ``tier-2-managed/`` package.

Mirrors the ``tier_1_naive`` shim from Phase 128. Hyphenated directory names
are not valid Python identifiers; this shim loads each sibling module from
``tier-2-managed/`` via ``importlib`` and registers them under
``tier_2_managed.<name>``.
"""
from __future__ import annotations
import importlib.util
import sys
from pathlib import Path

_TIER2_DIR = Path(__file__).resolve().parent.parent / "tier-2-managed"

_tier2_str = str(_TIER2_DIR)
if _tier2_str not in sys.path:
    sys.path.insert(0, _tier2_str)


def _load(submodule: str):
    full_name = f"{__name__}.{submodule}"
    if full_name in sys.modules:
        return sys.modules[full_name]
    src = _TIER2_DIR / f"{submodule}.py"
    if not src.exists():
        raise ImportError(f"tier_2_managed shim: missing source file {src}")
    spec = importlib.util.spec_from_file_location(full_name, src)
    if spec is None or spec.loader is None:
        raise ImportError(f"tier_2_managed shim: could not load spec for {src}")
    mod = importlib.util.module_from_spec(spec)
    sys.modules[full_name] = mod
    spec.loader.exec_module(mod)
    return mod


for _name in ("store", "query", "main"):
    try:
        _load(_name)
    except ImportError:
        # query/main may not exist yet during partial Plan 02 execution; skip.
        pass
```

2. Create `tier-2-managed/store.py` implementing the 5 lifecycle functions. Use the contract above. Full reference implementation:

   ```python
   """Tier 2 (Gemini File Search) — FileSearchStore lifecycle helpers.

   This module wraps `client.file_search_stores.*` and the long-running
   `upload_to_file_search_store` operation. It is the only place in Tier 2
   that knows about operation polling and 503 retry semantics — main.py and
   query.py treat the store as a black-box handle.
   """
   from __future__ import annotations

   import time
   from pathlib import Path
   from typing import Iterable

   from google import genai
   from google.genai import types

   STORE_ID_PATH = Path(__file__).resolve().parent / ".store_id"
   STORE_DISPLAY_NAME = "rag-arch-patterns-tier-2"
   POLL_INTERVAL_S = 2.0  # Pattern 2 — cookbook uses 1-3s; 2s is the comfortable middle.


   def get_or_create_store(client: genai.Client, reset: bool = False):
       """Resolve the cached FileSearchStore handle, or create a new one.

       Caches the server-assigned name in `STORE_ID_PATH` (gitignored) so
       repeated `--ingest` calls re-use the same store. With `reset=True`,
       deletes both the cached handle and any server-side store with the
       cached name, then falls through to create a fresh one.
       """
       if reset and STORE_ID_PATH.exists():
           cached = STORE_ID_PATH.read_text().strip()
           try:
               client.file_search_stores.delete(name=cached, config={"force": True})
           except Exception:
               pass  # Already gone or never existed; OK.
           STORE_ID_PATH.unlink()

       if STORE_ID_PATH.exists():
           cached = STORE_ID_PATH.read_text().strip()
           try:
               return client.file_search_stores.get(name=cached)
           except Exception:
               STORE_ID_PATH.unlink()  # Stale; fall through.

       store = client.file_search_stores.create(
           config=types.CreateFileSearchStoreConfig(
               display_name=STORE_DISPLAY_NAME,
           )
       )
       STORE_ID_PATH.parent.mkdir(parents=True, exist_ok=True)
       STORE_ID_PATH.write_text(store.name)
       return store


   def upload_pdf(client: genai.Client, store_name: str, pdf_path: str, display_name: str):
       """Upload a single PDF and poll until the long-running operation is done.

       Returns the completed Operation object (op.response holds the imported
       document handle). Raises if the operation reports an error in op.error.
       """
       op = client.file_search_stores.upload_to_file_search_store(
           file=pdf_path,
           file_search_store_name=store_name,
           config=types.UploadToFileSearchStoreConfig(display_name=display_name),
       )
       # Poll until done. Operation.done flips True when the embedding/index step finishes.
       while not (op := client.operations.get(op)).done:
           time.sleep(POLL_INTERVAL_S)
       if getattr(op, "error", None):
           raise RuntimeError(f"upload_to_file_search_store failed for {pdf_path}: {op.error}")
       return op


   def upload_with_retry(
       client: genai.Client,
       store_name: str,
       pdf_path: str,
       display_name: str,
       max_attempts: int = 3,
       base_wait_s: float = 30.0,
   ):
       """Wrap `upload_pdf` with 30s/60s/120s exponential backoff (Pitfall 2)."""
       last_exc: Exception | None = None
       for attempt in range(max_attempts):
           try:
               return upload_pdf(client, store_name, pdf_path, display_name)
           except Exception as e:
               last_exc = e
               if attempt == max_attempts - 1:
                   break
               wait = base_wait_s * (2 ** attempt)
               time.sleep(wait)
       assert last_exc is not None
       raise last_exc


   def list_existing_documents(client: genai.Client, store_name: str) -> set[str]:
       """Return display_names of documents already in the store.

       Used to make `--ingest` idempotent (Pitfall 2 resilience): if a 503 storm
       interrupts a previous run, we resume from where we left off.
       """
       out: set[str] = set()
       pager = client.file_search_stores.documents.list(
           file_search_store_name=store_name,
       )
       for doc in pager:
           dn = getattr(doc, "display_name", None)
           if dn:
               out.add(dn)
       return out


   def delete_store(client: genai.Client, store_name: str) -> None:
       """Force-delete the store and remove the cached `.store_id` sidecar."""
       try:
           client.file_search_stores.delete(name=store_name, config={"force": True})
       except Exception:
           pass
       if STORE_ID_PATH.exists():
           STORE_ID_PATH.unlink()
   ```
