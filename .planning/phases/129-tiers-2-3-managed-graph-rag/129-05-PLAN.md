```python
"""Tier 3 (LightRAG) ingest path — PDF→text→rag.ainsert."""
from __future__ import annotations
from pathlib import Path

import fitz  # pymupdf — Tier 3 owns this dep (Pitfall 11)


def extract_full_text(pdf_path: str) -> str:
    """Concatenate all page text with double-newline separators."""
    parts: list[str] = []
    with fitz.open(pdf_path) as doc:
        for page in doc:
            t = page.get_text("text") or ""
            if t.strip():
                parts.append(t)
    return "\n\n".join(parts)


async def ingest_corpus(rag, papers: list[dict], console, tracker, model: str) -> int:
    """Ingest all papers into the LightRAG graph. Returns the number of papers ingested.

    Idempotent: passes stable doc_id=paper_id; LightRAG's kv_store_doc_status.json
    deduplicates re-runs. Cost projection is printed BEFORE the loop starts so
    users can decide to abort.
    """
    from shared.config import get_settings
    dataset_root = Path(get_settings().dataset_root)

    n_papers = sum(
        1 for p in papers
        if (dataset_root / "papers" / p["filename"]).exists()
    )
    console.print(
        f"[cyan]Tier 3 ingest plan: ~{n_papers} papers, ~$1.00 estimated entity-extraction cost "
        f"via {model} (one-time; idempotent on re-run via kv_store_doc_status).[/cyan]"
    )

    ingested = 0
    for p in papers:
        pdf_path = dataset_root / "papers" / p["filename"]
        if not pdf_path.exists():
            continue
        text = extract_full_text(str(pdf_path))
        await rag.ainsert(text, ids=[p["paper_id"]])
        ingested += 1
        console.print(f"  inserted {p['paper_id']} ({ingested}/{n_papers}) — running cost ${tracker.total_usd():.4f}")
    console.print(
        f"[green]Tier 3 ingest complete: {ingested} papers in graph at lightrag_storage/tier-3-graph/.[/green]"
    )
    return ingested
```

query.py contracts (Pattern 5):

```python
"""Tier 3 query path — wraps rag.aquery with token tracking."""
from __future__ import annotations
from lightrag import QueryParam


async def run_query(rag, question: str, mode: str, model: str, tracker) -> tuple[str, int, int]:
    """Run a graph query. Returns (answer_text, total_input_tokens, total_output_tokens).

    LightRAG's per-query LLM calls are captured by the CostAdapter wired through
    build_rag (Plan 03). This function reads the aggregate from the tracker AFTER
    the query so callers can render in_tok / out_tok.
    """
    pre_in = sum(q["input_tokens"] for q in tracker.queries if q.get("kind") == "llm")
    pre_out = sum(q["output_tokens"] for q in tracker.queries if q.get("kind") == "llm")
    answer = await rag.aquery(question, param=QueryParam(mode=mode))
    post_in = sum(q["input_tokens"] for q in tracker.queries if q.get("kind") == "llm")
    post_out = sum(q["output_tokens"] for q in tracker.queries if q.get("kind") == "llm")
    return answer, max(0, post_in - pre_in), max(0, post_out - pre_out)
```

If Plan 03 Task 1's probe revealed Outcome C (no `token_tracker` support on 1.4.15), `run_query` cannot rely on per-call capture. Fallback for that case: estimate post-query in/out via `tiktoken cl100k_base` over `(question + answer)` — accuracy ±20% is acceptable for the demo cost line. The choice between probe-driven vs estimate-fallback is documented in Plan 03's SUMMARY; this plan reads it and picks the right shape.

main.py shape (verbatim from 129-RESEARCH.md Code Example B with --yes gate addition):

```python
"""Tier 3: Graph RAG (LightRAG) CLI entrypoint.

Usage
-----
End-to-end demo (CONFIRMS first ingest costs ~$1; pass --yes to skip prompt):

    python tier-3-graph/main.py --yes

Run ingest only (idempotent — re-runs near-free via kv_store_doc_status):

    python tier-3-graph/main.py --ingest --yes

Query an existing graph:

    python tier-3-graph/main.py --query "Your question?" --mode hybrid

Wipe the graph and rebuild (~$1, ~10min):

    python tier-3-graph/main.py --reset --ingest --yes

Cost & latency are printed via shared.display.
"""
from __future__ import annotations

import argparse
import asyncio
import shutil
import sys
import time
from pathlib import Path

_REPO_ROOT = Path(__file__).resolve().parent.parent
if str(_REPO_ROOT) not in sys.path:
    sys.path.insert(0, str(_REPO_ROOT))

from rich.console import Console  # noqa: E402

from shared.config import get_settings  # noqa: E402
from shared.cost_tracker import CostTracker  # noqa: E402
from shared.display import render_query_result  # noqa: E402
from shared.loader import DatasetLoader  # noqa: E402

from tier_3_graph.rag import build_rag, WORKING_DIR, DEFAULT_LLM_MODEL  # noqa: E402
from tier_3_graph.cost_adapter import CostAdapter  # noqa: E402
from tier_3_graph.ingest import ingest_corpus  # noqa: E402
from tier_3_graph.query import run_query  # noqa: E402

DEFAULT_QUERY: str = (
    "Comparing Lewis et al. 2020's RAG and Karpukhin et al. 2020's DPR, "
    "how does dense passage retrieval relate to RAG's non-parametric memory?"
)
DEFAULT_MODE: str = "hybrid"  # Open Q4 — works without reranker; --mode mix opt-in
VALID_MODES = ["naive", "local", "global", "hybrid", "mix"]


def _confirm_or_abort(prompt: str, yes: bool, console: Console) -> bool:
    if yes:
        console.print(f"[dim]{prompt}  [confirmed via --yes][/dim]")
        return True
    console.print(f"[yellow]{prompt}[/yellow]")
    try:
        answer = input("Continue? [y/N]: ").strip().lower()
    except EOFError:
        return False
    return answer == "y"


async def amain(args, console: Console) -> int:
    settings = get_settings()
    if settings.openrouter_api_key is None:
        console.print(
            "[red]OPENROUTER_API_KEY required for Tier 3 (LightRAG entity-extraction + answers + embeddings). "
            "Copy .env.example to .env and set your key from https://openrouter.ai/keys[/red]"
        )
        return 2
    # Make the key visible to LightRAG's openai-compat funcs (Plan 03 reads from os.environ).
    import os
    os.environ["OPENROUTER_API_KEY"] = settings.openrouter_api_key.get_secret_value()

    if args.reset and Path(WORKING_DIR).exists():
        if not _confirm_or_abort(
            f"--reset will wipe {WORKING_DIR}/ — next --ingest will re-extract entities (~$1, ~10 min).",
            args.yes, console,
        ):
            console.print("[yellow]Aborted by user.[/yellow]")
            return 0
        shutil.rmtree(WORKING_DIR)

    tracker = CostTracker("tier-3")
    adapter = CostAdapter(tracker, model=args.model)
    rag = build_rag(llm_token_tracker=adapter, model=args.model)
    await rag.initialize_storages()

    try:
        if args.ingest:
            # Pitfall 3 — cost-surprise gate on FIRST ingest only.
            already_built = Path(WORKING_DIR, "graph_chunk_entity_relation.graphml").exists()
            if not already_built and not _confirm_or_abort(
                f"First-time ingest extracts a knowledge graph from ~100 papers. "
                f"Estimated cost: ~$1.00 via {args.model} (one-time). Re-runs are near-free.",
                args.yes, console,
            ):
                console.print("[yellow]Aborted by user.[/yellow]")
                return 0
            loader = DatasetLoader()
            papers = loader.papers()
            await ingest_corpus(rag, papers, console=console, tracker=tracker, model=args.model)

        if args.query:
            t0 = time.monotonic()
            answer_text, in_tok, out_tok = await run_query(
                rag, args.query, mode=args.mode, model=args.model, tracker=tracker,
            )
            latency = time.monotonic() - t0
            render_query_result(
                query=args.query,
                chunks=[],  # LightRAG doesn't expose per-chunk citations same as Tier 1/2; answer-only is honest
                answer=answer_text,
                cost_usd=tracker.total_usd(),
                input_tokens=in_tok,
                output_tokens=out_tok,
                console_override=console,
            )
            console.print(f"[bold]Latency:[/bold] {latency:.2f}s  [dim]mode={args.mode}  model={args.model}[/dim]")
            persisted = tracker.persist()
            console.print(f"[dim]Cost JSON written to {persisted}[/dim]")
        return 0
    finally:
        await rag.finalize_storages()


def _build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        prog="tier-3-graph",
        description="Tier 3 (Graph RAG) — LightRAG knowledge-graph extraction + multi-mode retrieval.",
    )
    parser.add_argument("--ingest", action="store_true",
                        help="Build the knowledge graph (one-time ~$1, idempotent on re-run).")
    parser.add_argument("--query", type=str, default=None,
                        help="Run a query against the persisted graph.")
    parser.add_argument("--mode", type=str, default=DEFAULT_MODE, choices=VALID_MODES,
                        help=f"LightRAG query mode (default {DEFAULT_MODE}). "
                             "naive=vector-only, local=entity-neighborhood, global=community-summary, "
                             "hybrid=local+global, mix=hybrid+rerank (requires reranker).")
    parser.add_argument("--reset", action="store_true",
                        help=f"Wipe {WORKING_DIR} before --ingest (warning: re-ingest costs ~$1).")
    parser.add_argument("--model", type=str, default=DEFAULT_LLM_MODEL,
                        help=f"OpenRouter LLM slug for entity extraction + answers (default %(default)s).")
    parser.add_argument("--yes", action="store_true",
                        help="Skip cost-surprise confirmation prompts (for non-interactive use).")
    return parser


def main(argv=None) -> int:
    args = _build_parser().parse_args(argv)
    console = Console()

    no_flags = not (args.ingest or args.query or args.reset)
    if no_flags:
        args.ingest = True
        args.query = DEFAULT_QUERY

    return asyncio.run(amain(args, console))


if __name__ == "__main__":
    raise SystemExit(main())
```
