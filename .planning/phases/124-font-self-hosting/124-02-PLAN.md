    **Assertion 2 — GA regression guard on dist/index.html:**
    Read `dist/index.html`. Assert it contains all of:
    - `googletagmanager.com`
    - `google-analytics.com`
    - `Content-Security-Policy`

    Fail with a clear error if any are missing, e.g.: `[verify-no-google-fonts] FAIL — dist/index.html missing required preservation marker 'googletagmanager.com' — CSP edit likely over-aggressive`

    **Assertion 3 — Preload hints present and correctly formed:**
    Read `dist/index.html`. Use a regex (or simple string matching) to find all `<link>` tags with `rel="preload"` AND `as="font"`. Assert:
    - Exactly 2 such tags exist (neither more nor fewer)
    - Each has `type="font/woff2"`
    - Each has a `crossorigin` attribute (bare or `crossorigin="anonymous"` — both valid)
    - Each `href` matches the pattern `/fonts/<name>-<weight>.woff2` (e.g., `/fonts/dm-sans-400.woff2`, `/fonts/bricolage-800.woff2`)

    Capture the two `href` values into a list for Assertion 4.

    **Assertion 4 — Preload hrefs match @font-face src URLs (Pitfall 1 guard):**
    Astro bundles CSS into hashed chunks under `dist/_astro/*.css`. The hand-written `@font-face` rules from `src/styles/global.css` will appear there. Walk all `.css` files in `dist/_astro/` and concatenate their text. For each `href` captured in Assertion 3, assert that a matching `@font-face ... src: url('<href>') format('woff2')` rule exists in the concatenated CSS.

    Be flexible on quoting/whitespace:
    - Accept `url('/fonts/dm-sans-400.woff2')`, `url("/fonts/dm-sans-400.woff2")`, or `url(/fonts/dm-sans-400.woff2)`
    - Accept the url being on same line as `@font-face` opener or any line within the declaration block

    Simplest implementation: for each href, assert the concatenated CSS text contains `url('${href}')` OR `url("${href}")` OR `url(${href})`. The `@font-face` wrapper presence is implicit — if the URL is in the CSS, it's in a rule.

    **Template (use this as a starting structure):**

    ```javascript
    // scripts/verify-no-google-fonts.mjs
    // Build-time gate for Phase 124: Font Self-Hosting
    // Asserts:
    //   (1) No fonts.googleapis.com / fonts.gstatic.com in dist/ HTML/CSS/JS (except allowlisted
    //       FastAPI guide code-example page)
    //   (2) CSP in dist/index.html still allows GA (regression guard)
    //   (3) Exactly 2 preload <link rel=preload as=font> hints with crossorigin and /fonts/*.woff2
    //   (4) Each preload href matches a @font-face src URL in the bundled CSS (Pitfall 1 guard)
    //
    // See .planning/phases/124-font-self-hosting/124-RESEARCH.md
    import { readFileSync, readdirSync, statSync } from 'node:fs';
    import { join, relative } from 'node:path';

    const ROOT = process.cwd();
    const DIST = join(ROOT, 'dist');
    const FORBIDDEN = ['fonts.googleapis.com', 'fonts.gstatic.com'];

    // Paths (relative to dist/) where the forbidden strings are allowed as
    // literal code examples teaching CSP. Keep this list as narrow as possible.
    const ALLOWLIST_PATTERNS = [
      /^guides\/fastapi-production\/.*security-headers/,
      // ADD MORE PATTERNS ONLY IF NEW CODE-EXAMPLE PAGES ARE ADDED
    ];

    function isAllowlisted(relPath) {
      return ALLOWLIST_PATTERNS.some(re => re.test(relPath));
    }

    // ---- Assertion 1 ----
    const violations = [];
    const allowlistHits = [];
    function walk(dir) {
      for (const entry of readdirSync(dir)) {
        const full = join(dir, entry);
        const s = statSync(full);
        if (s.isDirectory()) { walk(full); continue; }
        if (!/\.(html|css|js)$/.test(entry)) continue;
        const body = readFileSync(full, 'utf8');
        const relPath = relative(DIST, full);
        for (const needle of FORBIDDEN) {
          if (body.includes(needle)) {
            if (isAllowlisted(relPath)) {
              allowlistHits.push({ file: relPath, needle });
            } else {
              violations.push({ file: relPath, needle });
            }
          }
        }
      }
    }
    walk(DIST);

    // ---- Assertion 2 ----
    const indexHtml = readFileSync(join(DIST, 'index.html'), 'utf8');
    const GA_MARKERS = ['googletagmanager.com', 'google-analytics.com', 'Content-Security-Policy'];
    const missingMarkers = GA_MARKERS.filter(m => !indexHtml.includes(m));

    // ---- Assertion 3 ----
    const PRELOAD_RE = /<link[^>]*\brel="preload"[^>]*\bas="font"[^>]*>/g;
    const preloadMatches = indexHtml.match(PRELOAD_RE) || [];
    const preloadIssues = [];
    const preloadHrefs = [];
    if (preloadMatches.length !== 2) {
      preloadIssues.push(`expected exactly 2 preload <link as=font>, found ${preloadMatches.length}`);
    }
    for (const tag of preloadMatches) {
      if (!/\btype="font\/woff2"/.test(tag)) preloadIssues.push(`missing type=font/woff2: ${tag}`);
      if (!/\bcrossorigin\b/.test(tag)) preloadIssues.push(`missing crossorigin: ${tag}`);
      const href = tag.match(/\bhref="([^"]+)"/)?.[1];
      if (!href || !/^\/fonts\/[a-z0-9-]+\.woff2$/.test(href)) {
        preloadIssues.push(`invalid or missing /fonts/<name>.woff2 href: ${tag}`);
      } else {
        preloadHrefs.push(href);
      }
    }

    // ---- Assertion 4 ----
    const cssDir = join(DIST, '_astro');
    let allCss = '';
    try {
      for (const entry of readdirSync(cssDir)) {
        if (entry.endsWith('.css')) {
          allCss += readFileSync(join(cssDir, entry), 'utf8') + '\n';
        }
      }
    } catch { /* _astro may not exist if build shape changes — surface as an error below */ }
    const hrefMismatches = [];
    for (const href of preloadHrefs) {
      const patterns = [`url('${href}')`, `url("${href}")`, `url(${href})`];
      if (!patterns.some(p => allCss.includes(p))) {
        hrefMismatches.push(`preload href ${href} has no matching @font-face src in dist/_astro/*.css — Pitfall 1 (double-download risk)`);
      }
    }

    // ---- Report ----
    const hasFailure = violations.length > 0 || missingMarkers.length > 0 || preloadIssues.length > 0 || hrefMismatches.length > 0;

    for (const hit of allowlistHits) {
      console.log(`[verify-no-google-fonts] INFO — allowlisted match in ${hit.file} ("${hit.needle}")`);
    }

    if (hasFailure) {
      console.error('[verify-no-google-fonts] FAIL');
      for (const v of violations) {
        console.error(`  Google Fonts URL leak: ${v.file} contains "${v.needle}"`);
      }
      for (const m of missingMarkers) {
        console.error(`  GA regression: dist/index.html missing required marker "${m}"`);
      }
      for (const issue of preloadIssues) {
        console.error(`  Preload issue: ${issue}`);
      }
      for (const mm of hrefMismatches) {
        console.error(`  URL mismatch: ${mm}`);
      }
      process.exit(1);
    }

    console.log(`[verify-no-google-fonts] PASS — 0 violations, ${preloadHrefs.length} preload hints (${preloadHrefs.join(', ')}), all @font-face src URLs matched`);
    ```
