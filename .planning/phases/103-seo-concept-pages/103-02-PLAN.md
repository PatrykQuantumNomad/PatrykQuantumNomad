    Create `src/pages/ai-landscape/[slug].astro` following the established pattern from `beauty-index/[slug].astro` and `db-compass/[slug].astro`.

    **getStaticPaths:**
    ```typescript
    import { getCollection } from 'astro:content';
    import graphData from '../../data/ai-landscape/graph.json';
    import { getNodeRelationships } from '../../lib/ai-landscape/schema';
    import { buildAncestryChain } from '../../lib/ai-landscape/ancestry';
    import { conceptUrl, ogImageUrl, AI_LANDSCAPE_BASE } from '../../lib/ai-landscape/routes';
    import type { AiNode } from '../../lib/ai-landscape/schema';

    export async function getStaticPaths() {
      const entries = await getCollection('aiLandscape');
      const nodesMap = new Map(entries.map(e => [e.data.slug, e.data]));
      const clusters = new Map(graphData.clusters.map(c => [c.id, c]));

      return entries.map((entry) => {
        const node = entry.data;
        const relationships = getNodeRelationships(node.slug, graphData.edges);
        const ancestry = buildAncestryChain(node.slug, nodesMap);
        const cluster = clusters.get(node.cluster);

        return {
          params: { slug: node.slug },
          props: {
            node,
            cluster,
            relationships,
            ancestry,
            nodesMap: Object.fromEntries(nodesMap),
          },
        };
      });
    }
    ```

    **Component imports:**

    ```typescript
    import AncestryBreadcrumb from '../../components/ai-landscape/AncestryBreadcrumb.astro';
    import RelatedConcepts from '../../components/ai-landscape/RelatedConcepts.astro';
    ```
