import type { APIRoute, GetStaticPaths } from 'astro';
import { getCollection } from 'astro:content';
import { generateLanguageOgImage } from '../../../lib/og-image';

export const getStaticPaths: GetStaticPaths = async () => {
  const languages = await getCollection('languages');
  return languages.map((entry) => ({
    params: { slug: entry.data.id },
    props: { language: entry.data },
  }));
};

export const GET: APIRoute = async ({ props }) => {
  const png = await generateLanguageOgImage(props.language);
  return new Response(png, {
    headers: {
      'Content-Type': 'image/png',
      'Cache-Control': 'public, max-age=31536000, immutable',
    },
  });
};

    ```typescript
    import type { APIRoute, GetStaticPaths } from 'astro';
    import { getCollection } from 'astro:content';
    import graphData from '../../../data/ai-landscape/graph.json';
    import { generateAiLandscapeOgImage } from '../../../lib/og-image';

    const clusters = new Map(graphData.clusters.map(c => [c.id, c]));

    export const getStaticPaths: GetStaticPaths = async () => {
      const nodes = await getCollection('aiLandscape');
      return nodes.map((entry) => ({
        params: { slug: entry.data.slug },
        props: {
          name: entry.data.name,
          cluster: entry.data.cluster,
          description: entry.data.whyItMatters,
        },
      }));
    };

    export const GET: APIRoute = async ({ props }) => {
      const { name, cluster: clusterId, description } = props as {
        name: string;
        cluster: string;
        description: string;
      };
      const cluster = clusters.get(clusterId);
      const clusterName = cluster?.name ?? 'AI Concept';
      const clusterColor = cluster?.darkColor ?? '#c44b20';

      const png = await generateAiLandscapeOgImage(name, clusterName, clusterColor, description);

      return new Response(png, {
        headers: {
          'Content-Type': 'image/png',
          'Cache-Control': 'public, max-age=31536000, immutable',
        },
      });
    };
    ```

    Create the directory `src/pages/open-graph/ai-landscape/` if it does not exist.

    **Key details:**
    - Use `whyItMatters` for the description (short, punchy, ideal for OG images)
    - Resolve cluster from graph.json to get name and darkColor
    - Fallback to '#c44b20' (site accent) if cluster not found
    - Cache-Control: immutable (same as all other OG endpoints)
  </action>
