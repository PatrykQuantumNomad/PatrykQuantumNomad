**Shared helpers (duplicated in both files OR extracted to `scripts/shared/vs-text.mjs` — your call):**

```javascript
import { readFile, readdir, mkdir, writeFile } from 'node:fs/promises';
import { resolve, join } from 'node:path';

const DIST_DIR = resolve(process.cwd(), 'dist/beauty-index/vs');
const REPORTS_DIR = resolve(process.cwd(), '.planning/reports');

export async function listVsPageFiles() {
  const entries = await readdir(DIST_DIR, { withFileTypes: true });
  return entries
    .filter(e => e.isDirectory())
    .map(e => join(DIST_DIR, e.name, 'index.html'))
    .sort(); // deterministic ordering
}

export async function extractArticleText(htmlPath) {
  const html = await readFile(htmlPath, 'utf-8');
  // Match the main <article> or <main> content region.
  // Prefer <article> if present, else <main>, else fall back to <body>.
  let match = html.match(/<article[^>]*>([\s\S]*?)<\/article>/i);
  if (!match) match = html.match(/<main[^>]*>([\s\S]*?)<\/main>/i);
  if (!match) throw new Error(`No <article> or <main> in ${htmlPath}`);
  // Strip tags + collapse whitespace
  return match[1]
    .replace(/<script[\s\S]*?<\/script>/gi, ' ')
    .replace(/<style[\s\S]*?<\/style>/gi, ' ')
    .replace(/<[^>]+>/g, ' ')
    .replace(/&nbsp;/g, ' ')
    .replace(/&amp;/g, '&')
    .replace(/&lt;/g, '<')
    .replace(/&gt;/g, '>')
    .replace(/&quot;/g, '"')
    .replace(/&#39;/g, "'")
    .replace(/\s+/g, ' ')
    .trim();
}

export function tokenize(text) {
  return text.toLowerCase().replace(/[^\p{L}\p{N}\s]/gu, ' ').split(/\s+/).filter(Boolean);
}

export function shingles(tokens, k = 5) {
  const set = new Set();
  for (let i = 0; i + k <= tokens.length; i++) set.add(tokens.slice(i, i + k).join(' '));
  return set;
}

export function jaccard(a, b) {
  if (a.size === 0 && b.size === 0) return 1;
  let inter = 0;
  for (const s of a) if (b.has(s)) inter++;
  const union = a.size + b.size - inter;
  return union === 0 ? 0 : inter / union;
}

export function mulberry32(seed) {
  let s = seed >>> 0;
  return () => {
    s = (s + 0x6D2B79F5) >>> 0;
    let t = s;
    t = Math.imul(t ^ (t >>> 15), t | 1);
    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  };
}
```

**Script 1: `scripts/verify-vs-wordcount.mjs`**

1. Call `listVsPageFiles()`.
2. For each file: `extractArticleText()` -> `tokenize()` -> record `{ path, wordCount: tokens.length }`.
3. Compute min, max, mean, median. Build histogram buckets (500-599, 600-699, ..., 1400+).
4. Collect any page with `wordCount < 500` into a `failures` array.
5. Write `.planning/reports/vs-wordcount-{YYYYMMDD-HHMM}.json` with `{ generatedAt, min, max, mean, median, histogram, failures }`. Create `.planning/reports/` if missing.
6. Print a concise stdout summary: `[VS-07] 650 pages: min=X max=Y mean=Z median=W. Failures: N.`
7. If `failures.length > 0`, print up to 10 failure paths and `process.exit(1)`. Else `process.exit(0)`.
8. Warn (stderr, but DO NOT fail) if any page > 1500 words — indicates a potential bug in the dimension-paragraph assembler per RESEARCH Pattern 7.

Expected result for Phase 122: all 650 pages > 500 words, per CONTEXT target of 900-1200. If a page fails, the script output gives enough signal to fix (which page, what word count).

**Script 2: `scripts/verify-vs-overlap.mjs`**

1. Call `listVsPageFiles()`.
2. Seed a PRNG with a fixed integer constant (`const SEED = 20260416;`). Use `mulberry32(SEED)` to deterministically shuffle the file list and take the first 20.
3. For each sampled file: extract article text, tokenize, build 5-gram shingle set.
4. Compute Jaccard for all C(20,2) = 190 pairs. Record `{ pathA, pathB, jaccard }` for each.
5. Compute max, mean, 95th percentile across the 190 pair values.
6. Write `.planning/reports/vs-overlap-{YYYYMMDD-HHMM}.json` with `{ generatedAt, seed, sampledFiles, max, mean, p95, topCollisions: top10 by jaccard desc }`.
7. Print stdout summary: `[VS-06] sampled 20 of 650 pages (seed={SEED}): max Jaccard=X.XXX, mean=Y.YYY, p95=Z.ZZZ.`
8. If `max > 0.40`, print the top 5 colliding pairs and `process.exit(1)`. Else `process.exit(0)`.

Both scripts are idempotent and pure — running twice on the same dist produces identical output modulo the timestamp in the filename. Never write into `dist/` (RESEARCH Pitfall 6 — would pollute Phase 123's sitemap lastmod).
