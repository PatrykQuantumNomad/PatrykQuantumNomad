    const readingTimeMap = new Map<string, string>();
    for (const page of guidePages) {
      const { remarkPluginFrontmatter } = await render(page);
      if (remarkPluginFrontmatter?.minutesRead) {
        readingTimeMap.set(page.data.slug, remarkPluginFrontmatter.minutesRead);
      }
    }

    const accent = guideMeta.data.accentColor ?? '#D97706';

      return (
        <Tag href={exists ? guidePageUrl(guideMeta.data.slug, ch.slug) : undefined} class="...card styles...">
          <span class="font-mono text-lg" style={`color: ${accent}`}>{String(i + 1).padStart(2, '0')}</span>

          {ch.description && <p class="text-sm text-[var(--color-text-secondary)]">{ch.description}</p>}
          <div class="mt-auto flex items-center gap-2">
            {readTime && <span class="text-xs font-mono text-[var(--color-text-secondary)]">{readTime}</span>}
            {!exists && <span class="text-xs bg-[var(--color-surface-alt)] px-2 py-0.5 rounded">Coming Soon</span>}
