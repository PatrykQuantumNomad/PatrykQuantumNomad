shared/config.py public surface (REQUIRED gemini_api_key per RESEARCH.md Pattern 5; lazy factory keeps import-time tests green):
```python
from functools import lru_cache
from pydantic import SecretStr, Field
from pydantic_settings import BaseSettings, SettingsConfigDict

class Settings(BaseSettings):
    model_config = SettingsConfigDict(
        env_file=".env",
        env_file_encoding="utf-8",
        extra="ignore",
    )
    # REQUIRED — RESEARCH.md Pattern 5 verbatim. ValidationError if missing.
    gemini_api_key: SecretStr = Field(..., alias="GEMINI_API_KEY")
    openai_api_key: SecretStr | None = Field(None, alias="OPENAI_API_KEY")
    s2_api_key: SecretStr | None = Field(None, alias="S2_API_KEY")
    default_chat_model: str = "gemini-2.5-flash"
    default_embedding_model: str = "gemini-embedding-001"
    dataset_root: str = "dataset"

@lru_cache(maxsize=1)
def get_settings() -> Settings:
    """Lazy factory. ValidationError raised here, not at module import time.
    Callers (shared.llm, shared.embeddings, smoke_test) invoke this when they
    actually need keys. Import-only tests can `from shared import config` without
    a populated .env."""
    return Settings()

# NOTE: do NOT define a module-level `settings = Settings()` singleton.
# That breaks `tests/smoke_test.py::test_imports` in fresh checkouts without .env.
# Callers use `get_settings()` and let Pydantic raise ValidationError on missing keys.
```
